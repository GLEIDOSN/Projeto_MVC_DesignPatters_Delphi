/*COMMENTS*/
CREATE TABLE COMMENTS
(POSTID	INTEGER	NOT NULL,
ID		INTEGER NOT NULL,
NAME	VARCHAR(200),
EMAIL	VARCHAR(100),
BODY	BLOB SEGMENT SIZE 1,
CONSTRAINT PK_COMMENTS PRIMARY KEY (POSTID,ID));

CREATE TABLE USUARIOS (
  CODIGO INTEGER NOT NULL,
  NOME VARCHAR(200),
  ENDERECO VARCHAR(250),
  NUMERO INTEGER,
  BAIRRO VARCHAR(50),
  CIDADE VARCHAR(50),
  CEP VARCHAR(10),
  UF VARCHAR(2),
  CPF VARCHAR(11),
  LOGIN VARCHAR(40),
  SENHA VARCHAR(20),
CONSTRAINT PK_CODIGO PRIMARY KEY (CODIGO));
